1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  export APP_HOST=$(kubectl get ingress {{ include "agno-agentos-starter.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath='{.spec.rules[0].host}')
  echo http://$APP_HOST/
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get service {{ include "agno-agentos-starter.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -n {{ .Release.Namespace }} -o jsonpath='{.items[0].status.addresses[0].address}')
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
    echo "Wait for the load balancer IP to be assigned, then use kubectl get svc {{ include "agno-agentos-starter.fullname" . }} -n {{ .Release.Namespace }}"
{{- else }}
    echo "Use kubectl port-forward svc/{{ include "agno-agentos-starter.fullname" . }} 8000:{{ .Values.service.port }} -n {{ .Release.Namespace }}"
{{- end }}
