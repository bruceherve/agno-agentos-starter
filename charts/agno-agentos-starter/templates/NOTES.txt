  API: http://{{ (index .Values.ingress.hosts 0).host }}/

  export NODE_PORT=$(kubectl get svc {{ include "agno-agentos-starter.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -n {{ .Release.Namespace }} -o jsonpath='{.items[0].status.addresses[0].address}')
  API: http://$NODE_IP:$NODE_PORT

  API: Run: kubectl get svc {{ include "agno-agentos-starter.fullname" . }} -n {{ .Release.Namespace }}
{{- else }}
  API port-forward: kubectl port-forward svc/{{ include "agno-agentos-starter.fullname" . }} 8000:{{ .Values.service.port }} -n {{ .Release.Namespace }}
